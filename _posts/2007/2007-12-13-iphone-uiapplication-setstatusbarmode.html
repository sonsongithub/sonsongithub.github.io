[app setStatusBarMode:mode orientation:degree duration:duration_time fenceID:0 animation:anim];

mode = 0 普通の灰色
mode = 1 黒の半透明
mode = 2 見えなくなるーinvisible
mode = 3 背景が透明
anim = 0 フェードインフェードアウト
anim = 1 上に上がってくる
anim = 2 下へ落ちる，何処かの角を軸に回って，下へ位置する
anim = 3 画面の上端からニュルっと出てくる
anim = 4 前のものが次の移動方向に動きながら，次のものがその方向からあがってくる
anim = 5 前のものが残ったまま，次のものがフェードイン
anim = 6 ・・・？
