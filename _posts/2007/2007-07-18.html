<script src="./prototype.js"></script>
<script>
objs3 = new Array();
function Area3( inputx, inputy, inputnumber ) {
var x = inputx;
var y = inputy;
var number = inputnumber;
this.setup = setup;
function setup() {
var area = "area" + number;
$('canvas20070718').innerHTML += "<div id='" + area + "'>"+area+"</div>\n"
$(area).style.position = "absolute";
$(area).style.left = x + "px";
$(area).style.top  = y + "px";
$(area).style.width  = 50 + "px";
$(area).style.height = 50 + "px";
$(area).style.backgroundColor = "#AAAAAA";
}
}
function clicked3(event) {
var obj = new Area3( Event.pointerX(event), Event.pointerY(event), objs.size() );
obj.setup();
objs3.push( obj );
}
Event.observe( 'canvas20070718', 'click', clicked3, false);
</script>
