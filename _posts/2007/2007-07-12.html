Areaというクラスで箱を描画しまくるスクリプト．

<div id="canvas" style="width:490px;height:300px;border-style:solid;"></div>
<script src="./prototype.js"></script>
<script>
objs = new Array();
function Area( inputx, inputy, inputnumber ) {
var x = inputx;
var y = inputy;
var number = inputnumber;
this.setup = setup;
function setup() {
area = "area" + number;
$('canvas').innerHTML += "<div id='" + area + "'>"+area+"</div>\n"
$(area).style.position = "absolute";
$(area).style.left = x + "px";
$(area).style.top  = y + "px";
$(area).style.width  = 50 + "px";
$(area).style.height = 50 + "px";
$(area).style.backgroundColor = "#AAAAAA";
}
}
$('canvas').onmousedown = function (event) {
obj = new Area(	event.pageX, event.pageY, objs.size() );
obj.setup();
objs.push( obj );
}
</script>

この上の枠内をクリックすると，IDを持った箱が表示されます．
データを後で利用していないのでオブジェクト指向でやる意味がないのかもしれませんが，これはこの後の複線です．
