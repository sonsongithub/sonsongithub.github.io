行列計算

自分でプログラミングすると，勉強になるだろうなぁ・・・．
と思いながら，実際に使うとなると自分が作ったものを
その信頼性故に，使いたくないものに分類される逆行列．
後，大学時代によくわからなかった固有値もできるみたいだ．
すげぇ簡単です，ハイ．

#include <iostream>
#include <octave/config.h>
#include <octave/Matrix.h>
using namespace std;
int main ( int argc, void **argv ) {
Matrix m(2, 2);
m(0,0) = 3; m(0,1) = 2;
m(1,0) = 4; m(1,1) = 5;
cout << "Original Matrix" << endl <<  m << endl;
// Eigenvalue vector & value
EIG eig(m);
cout << "Eigen Vectors" << endl <<  eig.eigenvectors() << endl;
cout << "Eigen Values" << endl <<  eig.eigenvalues() << endl;
// Inverse matrix
Matrix in = m.inverse();
cout << "Inverse Matrix" << endl <<  in << endl;
return 0;
}
