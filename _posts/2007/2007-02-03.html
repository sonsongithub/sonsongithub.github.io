このアルゴリズムの名前をバブルソートだと思ってた・・・無知すぎる・・・・orz

#include <stdio.h>
#include <stdlib.h>
void merge( int *a, int *b, int *c, int size) {
int a_count = size;
int b_count = size;
while( a_count && b_count ) {
if( *a < *b ) {
*c++ = *a++;
a_count--;
}
else {
*c++ = *b++;
b_count--;
}
}
while( a_count > 0 ) {
*c++ = *a++;
a_count--;
}
while( b_count > 0 ) {
*c++ = *b++;
b_count--;
}
}
int main( int argc, void ** argv ) {
#define ARRAY_NUM 8
int a[ARRAY_NUM] = { 3, 1, 4, 8, 7, 6, 2, 5 };
int b[ARRAY_NUM];
int*front = a;
int*back = b;
for( int j = 1; j < ARRAY_NUM; j*=2 ) {
for( int i = 0; i < ARRAY_NUM; i+=(j*2) ) {
merge( front+i, front+i+j, back+i, j );
}
int *temp = (int*)back;
back = front;
front = temp;
}
for( int i = 0; i < ARRAY_NUM; i++ )
printf( "%d ", front[i] );
return 0;
}
