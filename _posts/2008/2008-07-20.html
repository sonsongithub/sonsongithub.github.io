C++でハッシュっぽいことができるSTL．
ambee氏に「それくらい知っとけ，ボケ」と怒られたわけではないが，教えてもらって非常に便利だったのでモメ．
mapというテンプレートで実現できるらしい．
イテレータのfirstでキーにsecondで値にアクセスできる．
findで値を見つけようとしたときに戻り値がend場合，値が見つからなかったことを示す．

#include <stdio.h>
#include <map>
using namespace std;
int main( int argc, void **argv ) {
map<int, int> idPointMap;
idPointMap.insert(pair<int, int>( 10, 64));
idPointMap.insert(pair<int, int>( 1280, 98 ));
map<int, int>::iterator itr;
// id
itr = idPointMap.find( 10 );
if( itr == idPointMap.end() ) {
printf( "not found\n" );
}
else {
printf( "id->%d, point->%d\n", itr->first, itr->second );
}
return 0;
}
